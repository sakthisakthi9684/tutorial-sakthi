2.	git add .
3.	git commit -m "commit message>
4.	git push origin <branch name>
5.	git pull origin <branch name>
6.	git branch
7.	git branch <new branch name>
8.	git checkout <branch name>

10.	git status
11.	git remote -v

13.	git merge origin/main

18.	git stash pop
19.	git stash clear
20.	git stash push -m "My custom message"
